import React, { useState, useEffect } from 'react';
import Auth from './components/Auth';
import Navbar from './components/Navbar';
import AddExpense from './components/AddExpense';
import Dashboard from './components/Dashboard';
import Goals from './components/Goals';
import Tips from './components/Tips';
import ParentView from './components/ParentView';

function App() {
  const [user, setUser] = useState(null);
  const [currentView, setCurrentView] = useState('dashboard');
  const [privacyMode, setPrivacyMode] = useState(false);
  const [parentView, setParentView] = useState(false);

  // Load user from localStorage
  useEffect(() => {
    const storedUser = localStorage.getItem('zeniwise_user');
    if (storedUser) {
      setUser(JSON.parse(storedUser));
    }
  }, []);

  // Storage schema initialization
  useEffect(() => {
    if (!localStorage.getItem('zeniwise_data')) {
      localStorage.setItem('zeniwise_data', JSON.stringify({
        income: 0,
        expenses: [],
        goals: []
      }));
    }
  }, []);

  if (!user) {
    return <Auth setUser={setUser} />;
  }

  return (
    <div className="min-h-screen bg-gray-50">
      <Navbar 
        setCurrentView={setCurrentView} 
        privacyMode={privacyMode} 
        setPrivacyMode={setPrivacyMode}
        parentView={parentView}
        setParentView={setParentView}
      />
      <div className="container mx-auto p-4">
        {parentView ? (
          <ParentView />
        ) : (
          <>
            {currentView === 'add-expense' && <AddExpense />}
            {currentView === 'dashboard' && <Dashboard privacyMode={privacyMode} />}
            {currentView === 'goals' && <Goals />}
            {currentView === 'tips' && <Tips />}
          </>
        )}
      </div>
    </div>
  );
}

export default App;

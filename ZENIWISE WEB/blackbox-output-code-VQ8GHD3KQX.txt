import React, { useState } from 'react';

function Auth({ setUser }) {
  const [isLogin, setIsLogin] = useState(true);
  const [username, setUsername] = useState('');
  const [password, setPassword] = useState('');

  const handleSubmit = (e) => {
    e.preventDefault();
    // Simple auth: store in localStorage
    const user = { username, password };
    localStorage.setItem('zeniwise_user', JSON.stringify(user));
    setUser(user);
  };

  return (
    <div className="flex items-center justify-center min-h-screen">
      <div className="card w-full max-w-md">
        <h2 className="text-2xl font-semibold mb-4">{isLogin ? 'Login' : 'Sign Up'}</h2>
        <form onSubmit={handleSubmit}>
          <input
            type="text"
            placeholder="Username"
            value={username}
            onChange={(e) => setUsername(e.target.value)}
            className="w-full p-2 mb-2 border rounded"
            required
          />
          <input
            type="password"
            placeholder="Password"
            value={password}
            onChange={(e) => setPassword(e.target.value)}
            className="w-full p-2 mb-4 border rounded"
            required
          />
          <button type="submit" className="btn w-full">Submit</button>
        </form>
        <button onClick={() => setIsLogin(!isLogin)} className="mt-2 text-blue-500">
          {isLogin ? 'Need to sign up?' : 'Already have an account?'}
        </button>
      </div>
    </div>
  );
}

export default Auth;
